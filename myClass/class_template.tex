	\[
	\alpha x + \beta y + \gamma z = p.
	\]
	
	\emphc{Primer:} Imamo sistem treh ravnin
	
	\begin{equation} \label{eq:plane_system}
		\begin{split}
		4 x + 2 y + z & = 7\\
		2 x + y - z & = 5\\
		x + 2 y + 2 z & = 3
		\end{split}	
	\end{equation}
	
	kjer je potrebno poiskati skupno točko $T(x,y,z)$.
